第一步：函数(获取数据)

import urllib.request,urllib.error
def askUrl(url):
head={“user-agent”:”Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/98.0.4758.82 Safari/537.36”}
req=urllib.request.Request(url,headers=head)
html=””
try:
response=urllib.request.urlopen(req)
page=response.read()
html=page.decode(“utf-8”)
return html
except urllib.error.URLError as e:
if hasattr(e,”code”):
print(e.code)
if hasattr(e,”reason”):
print(e.reason)

涉及点：urllib模块、user-agent代理替换、访问封装、得到响应response、utf-8解码、捕获异常。

第二步：函数(逐一解析数据)

from bs4 import BeautifulSoup
import re
#索取连接
findLink= re.compile(r’‘)#创建正则表达式对象
#索取图片
findImg = re.compile(r’src=”(.?)”‘,re.S) #re.S直接忽视里面的换行符
#索取片名
findmoviename= re.compile(r’(.)‘)
#索取评分
findscore=re.compile(r’(.)‘)
#索取评价人数
findjudge=re.compile(r’(\d)人评价‘)
#找到概况
findsurvey=re.compile(r’(.)‘)
#找到电影有关类型
findBd=re.compile(r’

(.?)

‘,re.S

def getData(baseurl):
datalist= []
for i in range(0,10):
url=baseurl+str(i*25) #每页25个，循环10次则可以获取TOP250的数据
html=askURL(url) #保存每一页数据
#2.逐一解析数据
soup=BeautifulSoup(html,”html.parser”)
for item in soup.find_all(‘div’,class_=”item”):
data=[]
item=str(item) #替换成字符串形式

        link=re.findall(findLink,item)[0]           #re库用来通过正则表达式查找指定的字符串
        data.append(link)
        
        Img=re.findall(findImg,item)[0]             #添加图片
        data.append(Img)
        
        moviename=re.findall(findmoviename,item)
        if (len(moviename)==2):                     #添加中文电影名字
            c_moviename=moviename[0]
            data.append(c_moviename)
            o_moviename=moviename[1].replace("/","") #去掉无用的符号
            data.append(o_moviename)                 #添加外国名
        else:
            data.append(moviename[0])
            data.append(" ")                         #外国名字留空
        
        score=re.findall(findscore,item)[0]         #添加评分
        data.append(score)
        
        judge=re.findall(findjudge,item)[0]         #添加评论人数
        data.append(judge)
        
        survey=re.findall(findsurvey,item)          #添加概况
        if len(survey)!=0:
            survey=survey[0].replace("。","")       #去除概况中的。号
            data.append(survey)
        else:
            data.append(" ")                        #无则留空
        
        Bd=re.findall(findBd,item)[0]               #添加电影类型
        Bd=re.sub('<br(\s+)?>(\s+)?'," ",Bd)        #去掉<br/>
        Bd=re.sub('/',"",Bd)
        data.append(Bd.strip())   #去掉前后的空格
        
        datalist.append(data)                       #把处理好的一部电影信息放入datalist
return datalist

涉及点：页面数的循环解析->BeautifulSoup函数解析html->逐一字符串形式输出对应标签区域的信息->re库函数索引->re库函数查找对应字段并添加到data->保存至总列表

保存数据(Excel保存)

import xlwt
def saveData(datalist,savepath):
print(“sava…”)
book = xlwt.Workbook(encoding=”utf-8”,style_compression=0) #创建workbook对象
sheet=book.add_sheet(‘豆瓣电影Top250’,cell_overwrite_ok=True) #创建工作表
col = (“电影详情链接”,”图片链接”,”影片中文名”,”影片英文名”,”评分”,”评价数”,”概况”,”相关信息”)
for i in range(0,8):
sheet.write(0,i,col[i])#列名
for i in range(0,250):
print(“第%d条”%(i+1))
data=datalist[i]
for j in range(0,8):
sheet.write(i+1,j,data[j]) #数据

book.save(savepath)            #保存

涉及点：创建workbook对象->创建工作表->循环打印列名->循环打印数据->保存
